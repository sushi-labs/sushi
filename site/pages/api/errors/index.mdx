export const entries = Object.entries(
  import.meta.glob('./*.mdx', { as: 'raw', eager: true })
)
  .filter(([path]) => !path.endsWith('index.mdx'))
  .map(([path, raw]) => {
    const slug = path.replace('./', '').replace(/\.mdx$/, '')
    const h1 = raw.match(/^#\s+(.+)$/m)?.[1]?.trim()
    const table = raw.match(/\|https?:\/\/[^|]+\|([^|]+)\|([^|]+)\|/)
    const title = h1 ?? slug.replace(/-/g, ' ').replace(/\b\w/g, (c) => c.toUpperCase())
    const status = table?.[2]?.trim() ?? ''
    return { slug, title, status }
  })
  .sort((a, b) => a.title.localeCompare(b.title))
